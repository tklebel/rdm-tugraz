library(visdat)

loadd(data)
data %>% 
  select_if(~sum(is.na(.x)) > 10)  %>% 
  vis_miss(cluster = T, sort_miss = T) +
  theme(plot.margin = margin(r = 50))

data %>% 
  select_if(~sum(is.na(.)) > 10) %>% 
  select_if(!str_detect(names(.), "other|comment")) %>% 
  vis_miss(cluster = T, sort_miss = T) +
  theme(plot.margin = margin(r = 50))
